开发脚手架及封装自动化构建工作流

一. 工程化概述

1. 解决的问题
  • 使用ES6+时的兼容性问题
  • 使用Less/Sass/PostCSS增强CSS的编程性，但运行环境无法直接支持
  • 使用模块化的方式提高项目的可维护性，但运行环境无法直接支持
  • 重复性工作：部署上线前手动压缩代码及资源文件，部署过程需手动上传代码
  • 多人协同开发，无法硬性统一大家的代码风格，pull回来的代码质量无法保证
  • 部分功能需要等待后端接口服务
  • 整体依赖后端项目
2.表现
  • 创建项目->编码->预览/测试->提交->部署
3.工程化!==某个工具
  • 规划项目的整体结构
  • 一些脚手架相当于特定类型的项目官方给出的集成式工程化方案
4.工程化离不开Nodejs
  • 前端工程化由nodejs提供

二. 脚手架工具

  • 本质作用：创建项目基础结构、提供项目规范和约定
  • 相同的组织结构、开发范式、模块依赖、工具配置、基础代码
  • 前端脚手架：以独立的工具存在

  • 脚手架的工作原理：创建node-cil应用

    1. 新建node-cli项目：sample-scaffolding
    2. yarn init初始化
    3. 在package.json中增减脚手架入口文件配置: "bin": "cli.js",
    4. 新建cli.js文件：必须包含头：#!/usr/bin/env node
    5. 输入以下脚手架代码
        //注意安装需要的依赖
    const fs = require('fs')
    const path = require('path')
    const inquirer = require('inquirer')
    const ejs = require('ejs')

    inquirer.prompt([
    {
        type: 'input',
        name: 'name',
        message: 'Project name?'
    }
    ])
    .then(anwsers => {
    // console.log(anwsers)
    // 根据用户回答的结果生成文件

    // 模板目录
    const tmplDir = path.join(__dirname, 'templates')
    // 目标目录
    const destDir = process.cwd()

    // 将模板下的文件全部转换到目标目录
    fs.readdir(tmplDir, (err, files) => {
        if (err) throw err
        files.forEach(file => {
        // 通过模板引擎渲染文件
        ejs.renderFile(path.join(tmplDir, file), anwsers, (err, result) => {
            if (err) throw err

            // 将结果写入目标文件路径
            fs.writeFileSync(path.join(destDir, file), result)
        })
        })
    })
    })
    1. yarn link 到全局
    2. 运行sample-scaffolding；如果提示找不到命令，请查看yarn global bin 将此目录配置到环境变量，因为sample-scaffolding命令程序是放到此路径bin目录下的

  • 构建过程核心工作原理（基于流的构建系统）

        const fs = require('fs')
    const { Transform } = require('stream')

    exports.default = () => {
        //文件读取流
        const read = fs.createReadStream('xxx.css')
        //文件写入流
        const write = fs.createWriteStream('xxx.css')
        //文件转换流
        const transform = new Transform({
            transform:(chunk,encoding,callback) => {
                //核心转换过程实现
                //chunk => 读取流中读取到的内容（Buffer）
                const input = chunk.toString()
                const output = input.replace(/\s+/g,'').repalce(/\/\*.+?\*\//g,'')//去除空白字符和css注释
                callback(null,output)//错误优先
            }
        })
        //把读取出来的文件流导入
        read
        .pipe(transfrom) //转换
        .pipe(write) //写入
        return read
    }

  • Gulp案例-useref:处理外部引入模块下的文件依赖

        //使用gulp-useref处理引入依赖
    //必须compile形成构建注释之后才能执行useref
    //使用gulp-if:文件类型判断|gulp-htmlmin：html压缩|gulp-uglify:js压缩| gulp-clean-css：css压缩 等4个插件压缩
    const useref = () => {//必须compile形成构建注释之后才能执行，否则无法执行，因为它只识别构建注释
        return src('dist/*.html',{base:'dist'})
            .pipe(plugins.useref({searchPath:['dist','.']}))//查找dist和当前项目根目录下的构建注释做转化
            //html js css
            .pipe(plugins.if(/\.js$/,plugins.uglify()))
            .pipe(plugins.if(/\.css$/,plugins.cleanCss()))
            .pipe(plugins.if(/\.html$/,plugins.htmlmin({
                collapseWhitespace:true,  //默认不处理换行符，配置后会去除所有空白字符
                minifyCSS:true, //对style标签内样式做压缩
                minifyJS:true   //对script标签内脚本做压缩
                //还可删除注释和空属性，根据文档配置
            })))
            .pipe(dest('release'))  //更改文件目录避免冲突【但打破了目录结构，下节处理】
    }

  • Gulp案例-案例总结：设置中转目录和修正最终目录，指定需要暴露的模块->最终在package.json中scripts属性下配置所导出的命令

        const {src,dest,parallel,series,watch} = require('gulp')
    
    const del = require('del')

    const browserSync = require('browser-sync') 

    const loadPlugins = require('gulp-load-plugins')
    const plugins = loadPlugins()
    
    const bs = browserSync.create()

    const data = {
        
    }

    const clean = () => {
        return del(['dist','temp'])//添加临时目录，作为useref的中转
    }

    const style = () => {
        return src('src/assets/styles/*.scss',{base:'src'}) 
            .pipe(plugins.sass({outputStyle:'expanded'})) 
            .pipe(dest('temp'))//临时目录
            .pipe(bs.reload({stream:true}))
    }

    const script = () => {
        return src('src/assets/scripts/*.js',{base:'src'}) 
            .pipe(plugins.babel({presets:['@babel/preset-env']})) 
            .pipe(dest('temp'))//临时目录
            .pipe(bs.reload({stream:true}))
    }

    const page = () => {
        return src('src/*.html',{base:'src'}) 
            .pipe(plugins.swig({data,defaults: { cache: false }}))//防止因缓存无法刷新法热更新
            .pipe(dest('temp'))//临时目录
            .pipe(bs.reload({stream:true}))
    }

    const image = () => {
        return src('src/assets/images/**',{base:'src'}) 
            .pipe(plugins.imagemin())  
            .pipe(dest('dist'))
    }

    const font = () => {
        return src('src/assets/fonts/**',{base:'src'}) 
            .pipe(plugins.imagemin())  
            .pipe(dest('dist'))
    }

    const extra = () => {
        return src('public/**',{base:'public'})
            .pipe(dest('dist'))
    }

    const serve = () => {
        watch('src/assets/styles/*.scss',style) 
        watch('src/assets/scripts/*.js',script)
        watch('src/*.html',page)
        watch([
            'src/assets/images/**',
            'src/assets/fonts/**',
            'public/**'
        ],bs.reload)
        bs.init({zhi
            notify:false, 
            port:2080,
            server:{
                baseDir:['temp','src','public'], //临时目录
                routes:{//开发环境读取依赖里的样式
                    '/node_modules':'node_modules'
                }
            }
        })
    }

    const useref = () => {
        return src('temp/*.html',{base:'temp'})//临时目录
            .pipe(plugins.useref({searchPath:['temp','.']}))//临时目录
            //html js css
            .pipe(plugins.if(/\.js$/,plugins.uglify()))
            .pipe(plugins.if(/\.css$/,plugins.cleanCss()))
            .pipe(plugins.if(/\.html$/,plugins.htmlmin({
                collapseWhitespace:true,  
                minifyCSS:true, 
                minifyJS:true
            })))
            .pipe(dest('dist'))  //改回原目录,另设中转目录
    }
    const compile = parallel(style,script,page)

    //上线之前执行的任务
    const build = series(clean,
        parallel(
            series(compile,useref),//必须限compile生成构建注释才能useref
            image,
            font,
            extra
        )
    )

    //生成环境执行的任务【生产环境不需对类型不变的文件执行转化，在运行服务器时直接把路径映射到源代码路径】
    const develop = series(compile,serve)

    module.exports = {
        clean,
        build,
        develop
    }
