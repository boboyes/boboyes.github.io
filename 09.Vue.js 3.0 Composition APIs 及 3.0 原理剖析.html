1. Vue.js 3.0 不同的构建版本

  • cjs: commonjs规范
  • prod： 表示生产版本，已经压缩，反之未压缩
  • global: 表示全局，可在浏览器直接通过script标签引入，生成全局vue对象
  • runtime: 只包含运行时
  • esm: 浏览器支持的esm 模块导入方式，通过script标签属性type="module"
    • 注意： 此种方式导入模块执行会在dom树创建完成之后，DOMContentLoaded之前
  • bundler: 核心模块会被打包，除此之外只打包使用到的代码，让vue构建体积更小

  import { createApp, reactive, onMounted, onUnmounted } from 'vue'
  function createPostion(){

4. watch
  • watch的三个参数
    • 第一个参数： 要监听的数据
    • 第二个参数： 监听到数据变化后执行的函数，这个函数有两个参数分别是新值和旧值
    • 第三个参数： 选项对象，deep和immediate
  • watch的返回值
    • 取消监听的函数
  • 示例：
  //https://www.yesno.wtf/api 随机返回yes or no
  第一个参数不是字符串，是ref或reactive返回的字符串或对象
  createApp({
    setup(){
      const question = ref('')
      const anwser = ref('')
      watch(question,(newVal,oldVal)=>{
        const response = await fetch('https://www.yesno.wtf/api')
        const data = await response.json()
        anwser.value = data.answer
      })
      return {
        question,
        anwser
      }
    }
  }).$mount('#app')
5. watchEffect
  • watch函数的简化版本，也用于监听数据的变化
  • 接收一个函数作为参数，监听函数内响应式数据的变化
  <button @click="increase">increase</stop>
  <button @click="stop">stop</stop>
  createApp({
    setup(){
      const count = ref(0)
      const stop = watchEffect(()=>{  //返回一个函数，用于取消watch的监视
        console.log(count.value)
      })
      return {
        count,
        stop,
        increase: () => {
          count.value ++
        }
      }
    }
  }).$mount('#app')

2. Proxy
  //1. set和deleteProperty 中需要返回布尔类型的值
  //   严格模式下，如果返回false的话会出现Type Error的异常
  const target = {
    foo: 'xxx',

  const proxy = new Proxy(target, {

  const obj = {

      console.log(this)
      return this.bar
    }
  }
  const proxy = new Proxy(obj, {
    get (target, key, receiver) {
      if (key === 'bar') {
        return 'value - bar'
      }
      return Reflect.get(target, key, receiver)
    }
  })

3. reactive
  • 接收一个参数， 判断这参数是否是对象
  • 创建拦截器对象 handler, 设置get/set/deleteProperty
  • 返回Proxy对象
const isObject = val => val !== null && typeof val === 'object'
const convert = target => isObject(target) ? reactive(target) : target
const hasOwnProperty = Object.prototype.hasOwnProperty
const hasOwn = (target, key) => hasOwnProperty.call(target, key)
export function reactive(target) {
  if(!isObject(target)) return target

  const handler = {
    get (target, key, receiver) {
      //收集依赖
      track(target, key)

      const result = Reflect.get(target, key,receiver)
      return covert(result)
    },
    set (target, key, value, receiver) {
      let oldValue = Reflect.get(target, key, receiver)
      let result = true
      if (oldValue !== value) {
        result = Reflect.set(target, key, value, receiver)
        //触发更新
        trigger(target, key)
      }
      return result
    },
    deleteProperty (target, key) {
      const hadKey = hasOwn(target, key)
      const result = Reflect.deleteProperty(target, key)
      if (hadKey && result) {
        //触发更新
        trigger(target, key)
      }
      return result
    }
  }
  return new Proxy(target, handler)
}
let activeEffect = null
export function effect (callback) {
  activeEffect = callback
  callback()
  activeEffect = null
}
let targetMap = new WeakMap()
//依赖收集 (收集effect函数的依赖)
export function track (target, key) {
  if (!activeEffect) return
  let depsMap = targetMap.get(target)
  if (!depsMap) {
    targetMap.set(target, (depsMap = new Map()))
  }
  let dep = depsMap.get(key)
  if (!dep) {
    depsMap.set(key, (dep = new Set()))
  }
  dep.add(activeEffect)
}
//触发更新
export function trigger(target, key) {
  const depsMap = targetMap.get(target)
  if (!depsMap) return
  const dep = depsMap.get(key)
  if (dep) {
    dep.forEach(effect => {
      effect()
    })
  }
}
//创建响应式对象ref
export function ref (raw) {
  //判断raw是否是ref创建的对象，如果是的话直接返回
  if (isObject(raw) && raw.__v_isRef) return
  let value = convert(raw)
  const r = {
    __v_isRef: true,
    get value(){
      //收集依赖
      track(r, 'value')
      return value
    },
    set value(newValue) {
      if (newValue !== value) {
        raw = newValue
        value = convert(raw)
        trigger(r, 'value')
      }
    }
  }
  return r
}
//toRef 把reactive返回的对象属性转换成ref格式的响应式对象
export function toRef (proxy) {
  const ret proxy instanceof Array ? new Array(proxy.length) : {}

  for (const key in proxy) {
    ret[key] = toProxyRef(proxy, key)
  }
  return ret
}
function toProxyRef (proxy, key) {
  const r = {
    __v_isRef: true,
    get value () {
      return proxy[key]  //此处proxy是响应式对象，访问其属性不必再次收集依赖
    }，
    set value () {
      proxy[key] = newValue
    }
  }
  return r
}

export function computed (getter) {
  const result = ref()

  effect(() => (result.value = getter()))  //getter执行时其函数体内部的响应式属性会收集依赖

  return result
}
4. Vite

  • Vite是一个面向现代浏览器的一个更轻、更快的Web应用开发工具

  • 它基于ECMAScript标准原生模块系统（ES Modules) 实现

  • 开发依赖： Vite|命令工具 @vue/compiler-sfc|单文件组件编译

  • vite serve|开发服务器 vite build|打包命令

  • vite serve 不需要打包，直接开启服务器，在服务器端编译并返回结果给浏览器 按需即时编译

1. 特性
  • 快速冷启动
  • 模块热更新
  • 按需编译
  • 开箱即用
2. 优点和区别（webpack）
  • Vite:
    • HMR : 立即编译当前所修改的文件
    • 打包: 生产模式下打包 vite build命令，使用rollup打包，原生的动态导入实现代码切割，打包结果只支持现代浏览器
    • 不需要额外配置，默认支持TS(内置支持)，支持less/sass/stylus/postcss-内置支持（需要单独安装） JSX
  • webpack:
    • HMR : 会自动以这个文件为入口重写build一次，所有设计到的依赖都会被加载一遍
    • 打包： 所有模块打包到一个bundle中，浏览器环境不支持模块化，零散的模块文件产生大量的http请求
3. Vite核心功能
  • 静态Web服务器
  • 编译单文件组件： 拦截浏览器不识别的模块，并处理
  • HMR
4. 开发
  1. 静态Web服务器
  #!/usr/bin/env node
  const path = require('path')
  const { Readable } = require('stream')
  const Koa = require('koa')
  const send = require('koa-send')
  const compilerSFC = reuqire('@vue/compiler-sfc') //vue提供的单文件组件编译模块

  const app = new Koa()

  const streamToString = stream => new Promise((resolve,reject) => {
    const chunks = []
    stream.on('data', chunk => chunks.push(chunk))
    stream.on('end', () => resolve(Buffer.concat(chunks).toString('utf-8')))
    stream.on('error', reject)
  })

  const stringToStream = string => {
    const stream = new Readable()
    stream.push(string)
    stream.push(null)
    return stream
  }
  //3. 加载第三方模块
  app.use(async (ctx, next) => {
    //ctx.path --> /@modules/vue
    if(ctx.path.startsWith('/@modules/')) {
      const moduleName = ctx.path.substr(10)
      const pkgPath = path.join(process.cwd(), 'node_modules', moduleName, 'package.json')
      const pkg = require(pkgPath)
      ctx.path = path.join('/node_modules', moduleName, pkg.module)
    }
    await next()
  })

  //1. 静态文件服务器
  app.use(async (ctx, next) => {
    await send(ctx, ctx.path, { root: process.cwd(), index: 'index.html'})
    await next()
  })
  //4. 处理单文件组件
  app.use(async (ctx,next) => {
    if(ctx.path.endsWith('.vue')) {
      const contents = await streamToString(ctx.body)
      const { descriptor } = compilerSFC.parse(contents)
      let code
      if (!ctx.query.type) {
        code = descriptor.script.content
        code = code.replace(/export\s+default\s+/g, 'const __script = ')
        code += `
          import { render as __render } from "${ctx.path}?type=template"
          __script.render = __render
          export default __script
        `
      } else if (ctx.query.type === 'template') {
        const templateRender = compilerSFC.compilerTemplate({ sourse: descriptor.template.content })
        code = templateRender.code
      }
      ctx.type = 'application/javascript'
      ctx.body = stringToStream(code)  
    }
    await next()
  })

  //2. 修改第三方模块的路径
  app.use(async (ctx, next) => {
    if(ctx.type === 'application/javascript'){
      const contents = await streamToString(ctx.body)
      // import vue from 'vue'
      ctx.body = contents
        .replace(/(from\s+['"])(?![\.\/])/g, '$1/@modules/')
        .replace(/process\.env\.NODE_ENV/g, '"development"')
    }
  })

  app.listen(3000)
  console.log('Server running @ http://localhost:3000')
