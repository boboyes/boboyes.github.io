1. React Hooks 介绍

1. 基本
  • 16.8以后新增的功能
  • 对函数组件进行增强，让函数组件可以存储状态，可以拥有处理副作用的能力
  • 不使用类组件的情况下可以实现相同的功能，类组件会被逐渐废弃支持
  • 副作用代码： 只要不是把数据转换为视图的代码称为副作用代码
2. 类组件的不足
  • 缺少逻辑复用机制
  • 类组件会变得复杂难以维护，同一个业务逻辑被分散到多个生命周期函数中
  • 类成员方法不能保证this指向的正确性
3. hooks使用
  • useState()
    1. 设置状态值方法的参数可以是一个值也可以是一个函数
    2. 设置状态值方法的方法本身是异步的
  • useEffect()
    • 第二个参数 useEffect(() => {}, [count]) 只有指定数据发生变化时触发effect
    • useEffect中的参数函数不能是异步函数，因为useEffect函数要返回清理资源的函数，如果是异步函数就变成了返回Promise
    • 使用立即执行函数解决异步问题：
      useEffect(() => { 
    (async () => { 
      await axios.get() 
    })() 
  })
  • useReducer()
  • useRef()
    • 获取Dom元素对象
    • 即使组件重新渲染，保存的数据仍然还在，数据更改不会触发组件重新渲染
  • useCallback()
    • 性能优化，缓存函数，使组件重新渲染时得到相同的函数实例， 避免传递给里层组件的实例方法发生改变而导致重新渲染
  • useContext()
    • 跨组件层级获取数据时简化获取数据的代码
  • useMemo() 类似计算属性
    • 类似Vue的计算属性，可以监测某个值的变化，根据变化值计算新值
    • 会缓存计算结果，如果检测值不变，即使组件重新渲染也不会重新计算。 避免在每个渲染进行昂贵的计算
  • memo方法
    • 性能优化，如果本组件中的数据没有发生变化，阻止组件更新，类似组件中的PureComponent和shouldComponentUpdate
4. 自定义hook
  • 是标准的封装和共享逻辑的方式
  • 是一个函数，名称以use开头
  • 就是逻辑和内置HOOK的组合

6. 一些钩子的实现原理
  • useState钩子函数的实现原理
  let stateArr = [],setters = [], stateIndex = 0;

  function createSetter (index) {
    return function (newState) {
      stateArr[index] = newState
      render()
    }
  }

  function useState (initialState) {  
    stateArr[stateIndex] = stateArr[stateIndex] ? stateArr[stateIndex] : initialState
    setters.push(createSetter(stateIndex))
    let value = stateArr[stateIndex]
    let setter = setters[stateIndex]
    stateIndex ++ 
    return [value,setter]
  }

  function render() {
    stateIndex = 0;
    ReactDOM.render(<App />,document.getElementById('root'))
  }

  function App() {
    const [count,setCount] = useState(0)
    const [state,setState] = useState({name:'张三'})
    // debugger
    // stateIndex = 0
    return (
      <div className="App">
        <span>{count}</span>
        <button onClick={()=>setCount(count+1)}>+1</button>
        <span>{state.name}</span>
        <button onClick={()=>setState({name:'李四'})}>+1</button>
      </div>
    );
  }

6.自定义表单控件





    

  • 构建自定义表单控件复选框
    • 



import React from "react";
import { Formik, Form, Field, ErrorMessage, useField } from "formik";
import * as Yup from "yup";

function Checkbox ({label, ...props}) {
  const [field, meta, helper] = useField(props);
  const { value } = meta;
  const { setValue } = helper;
  const handleChange = () => {
    const set = new Set(value);
    if (set.has(props.value)) {
      set.delete(props.value);
    }else {
      set.add(props.value);
    }
    setValue([...set])
  }
  return <div>
    <label htmlFor="">
      <input checked={value.includes(props.value)} type="checkbox" {...props} onChange={handleChange}/> {label}
    </label>
  </div>
}

function App() {
  const initialValues = {username: '', hobbies: ['足球', '篮球']};
  const handleSubmit = (values) => {
    console.log(values);
  };
  const schema = Yup.object({
    username: Yup.string()
      .max(15, "用户名的长度不能大于15")
      .required("请输入用户名"),
  });
  return (
    <Formik
      initialValues={initialValues}
      onSubmit={handleSubmit}
      validationSchema={schema}
    >
      <Form>
        <Field name="username" />
        <ErrorMessage name="username" />
        <Checkbox value="足球" label="足球" name="hobbies"/>
        <Checkbox value="篮球" label="篮球" name="hobbies"/>
        <Checkbox value="橄榄球" label="橄榄球" name="hobbies"/>
        <input type="submit"/>
      </Form>
    </Formik>
  );
}

export default App;
